## 一、安装Nginx

* ### Docker安装

```shell
docker run --name nginx -d \
-v /mydata/nginx/conf/nginx.conf:/etc/nginx/nginx.conf \
-v /mydata/nginx/html:/usr/share/nginx/html \
-p 80:80 \
nginx:1.22
```

## 二、配置文件nginx.conf

```
worker_processes  auto;

error_log  /var/log/nginx/error.log notice;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile        on;
    #tcp_nopush     on;

    keepalive_timeout  65;

    #gzip  on;

    server {
	    listen 80 default_server;
        server_name _;
   
		return 404;
	}
	
    server {
        listen       80; #nginx要监听80端口
        server_name  www.zyg.com; #请求中携带的域名如果是www.zyg.com，就会走这个server

        # Load configuration files for the default server block.
        #include /etc/nginx/default.d/*.conf;

        location /login { #url中携带的路径必须和location匹配
            proxy_pass http://192.168.154.130:8080; #当前server所代理的服务
			root /html;
            index index.html;
			fastcgi_buffers 256 128k;
			chunked_transfer_encoding off; 
        }
    }
	
	    server {
        listen       80;
        server_name  www.zyg2.com;
		root /mydata/nginx/html;
		index index.html;

        # Load configuration files for the default server block.
        #include /etc/nginx/default.d/*.conf;

        location /login2 {
            proxy_pass http://192.168.154.130:8080;
			root /html;
            index index.html;
			fastcgi_buffers 256 128k;
			chunked_transfer_encoding off; 
        }
    }
}
```
